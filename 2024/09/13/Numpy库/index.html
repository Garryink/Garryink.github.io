import numpy as np
import time

matrix_sample = np.random.randint(2, 10, (3, 4))

matrix_sample

matrix_sample[2, 0]  # 读取3行1列元素
matrix_sample[:, 1]  # 读取第2列所有元素
matrix_sample[2, :]  # 读取第3行所有元素
matrix_sample[[0, 2]]  # 读取第1、3行的元素
matrix_sample[[1, 0]]  # 读取第2、1行的元素（顺序读取，根据位置需求读取）
matrix_sample[[0, 2], [1, 3]]  # 读取这两个位置的元素

matrix_temp2 = matrix_temp1[[0, 2, 1]]
matrix_temp2[2] = matrix_temp2[2] + matrix_temp2[1] * 3
matrix_temp2[0] = matrix_temp2[0] + matrix_temp2[1]

np.array([1,2,3])  # 一维数组

np.where(a>0.5)  # 返回数组中大于0.5的元素的索引

np.max(a)  # 数组元素求最大值
np.min(a)  # 数组元素求最小值
np.argmax(a)  # 数组元素求最大值的索引
np.argmin(a)  # 数组元素求最小值的索引
np.sort(a)  # 数组元素排序
np.argsort(a)  # 数组元素排序的索引
np.unique(a)  # 数组元素去重
np.concatenate((a,b),axis=0)  # 数组拼接
np.vstack((a,b))  # 数组垂直拼接
np.hstack((a,b))  # 数组水平拼接
np.dot(a,b)  # 数组点乘
np.linalg.inv(a)  # 数组求逆
np.linalg.det(a)  # 数组求行列式
np.linalg.eig(a)  # 数组求特征值和特征向量
